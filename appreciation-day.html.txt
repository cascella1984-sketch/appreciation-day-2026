<script>
const totalLikes = 250;
const tilesCount = 16;
let likes = 0;
let revealed = 0;
const size = 4;

const imageUrl = "appreciation.jpg";

const grid = document.getElementById("grid");
const counter = document.getElementById("likeCounter");
const button = document.getElementById("likeBtn");
const flipSound = document.getElementById("flipSound");

let tiles = [];

for(let i=0;i<tilesCount;i++){
  let tile = document.createElement("div");
  tile.classList.add("tile");

  let inner = document.createElement("div");
  inner.classList.add("inner");

  let front = document.createElement("div");
  front.classList.add("front");
  front.innerHTML="üçü";

  let back = document.createElement("div");
  back.classList.add("back");
  back.style.backgroundImage=`url(${imageUrl})`;
  back.style.backgroundPosition =
  `-${(i%size)*80}px -${Math.floor(i/size)*80}px`;

  inner.appendChild(front);
  inner.appendChild(back);
  tile.appendChild(inner);
  grid.appendChild(tile);
  tiles.push(inner);
}

button.addEventListener("click",()=>{
  if(likes>=totalLikes) return;

  likes++;
  counter.innerHTML = `üëç Like: ${likes} / ${totalLikes}`;

  if(revealed < tilesCount){
    tiles[revealed].classList.add("flipped");
    flipSound.play();
    revealed++;
  }

  if(likes === totalLikes){
    document.getElementById("finalMessage").style.display="block";
    launchConfetti();
    button.disabled=true;
  }
});

function launchConfetti(){
  const canvas = document.getElementById("confetti");
  const ctx = canvas.getContext("2d");
  canvas.width = window.innerWidth;
  canvas.height = window.innerHeight;

  let pieces = [];
  for(let i=0;i<200;i++){
    pieces.push({
      x: Math.random()*canvas.width,
      y: Math.random()*canvas.height,
      r: Math.random()*6+4,
      d: Math.random()*200
    });
  }

  function draw(){
    ctx.clearRect(0,0,canvas.width,canvas.height);
    ctx.fillStyle="#FFC72C";
    pieces.forEach(p=>{
      ctx.beginPath();
      ctx.arc(p.x,p.y,p.r,0,Math.PI*2,true);
      ctx.fill();
    });
  }

  draw();
}
